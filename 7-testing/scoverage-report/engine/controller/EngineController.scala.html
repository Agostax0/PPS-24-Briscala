20 object EventParser:
21   /**
22    * Parses a game event string into a GameEvent.
23    *
24    * @param event The event string in the format "playerName::cardInfo", with 'cardInfo' being
25    *  'name rank suit'.
26    * @return Either an error message or a CardPlayedEvent.
27    */
28   def parseEvent(event: String): Either[String, GameEvent] =
29     event.split("::").toList match
30       case playerName :: cardInfo :: Nil =>
31         parseCard(cardInfo).map(card => CardPlayedEvent(playerName, card))
32       case _ =>
33         Left(s"Invalid event format: expected 'playerName::cardInfo', got '$event'")
34 
35   private def parseCard(cardInfo: String): Either[String, CardModel] =
36     cardInfo.split(" ").toList match
37       case name :: rankStr :: suit :: Nil =>
38         rankStr.toIntOption match
39           case Some(rank) => Right(CardModel(name, rank, suit))
40           case None => Left(s"Invalid rank: $rankStr")
41       case _ =>
42         Left(s"Invalid card format: expected 'name rank suit', got '$cardInfo'")
