66 /** Selects a card based on the play rules of the game, aiming to win the current
67  * turn if possible. If no winning card is available, it selects the card
68  * with the lowest point.
69  * @param self the PlayerModel representing the bot itself
70  */
71 class RuleAwareDecisionStrategy(private val self: PlayerModel)
72     extends BotDecisionStrategy:
73   override def selectCard(
74       playerHand: DeckModel,
75       cardsOnTable: List[(PlayerModel, CardModel)],
76       gameContext: GameContext
77   ): CardModel =
78     val validCards = getValidCards(playerHand, cardsOnTable, gameContext)
79 
80     if cardsOnTable.isEmpty then selectOpeningCard(validCards, gameContext)
81     else selectWinningCard(validCards, cardsOnTable, gameContext)
82 
83   private def selectOpeningCard(
84       playerHand: List[CardModel],
85       context: GameContext
86   ): CardModel = playerHand.minBy(_.rank)
87 
88   private def selectWinningCard(
89       validCards: List[CardModel],
90       cardsOnTable: List[(PlayerModel, CardModel)],
91       context: GameContext
92   ): CardModel =
93     val expectedWinner = (card: CardModel) =>
94       context.calculateTurn(cardsOnTable :+ (self, card))
95     (
96       for
97         playableCard <- validCards
98         if expectedWinner(playableCard).get == self
99       yield playableCard
100     ).minByOption(card => context.calculatePoints(cardsOnTable :+ (self, card)))
101       .getOrElse(
102         validCards.minBy(card =>
103           context.calculatePoints(cardsOnTable :+ (self, card))
104         )
105       )
