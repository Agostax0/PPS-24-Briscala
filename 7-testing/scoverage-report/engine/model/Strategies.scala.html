67 /** Default implementation of PlayRuleStrategy that determines the winning
68   * player based on common briscola rules (highest card of the first suit)
69   */
70 class DefaultPlayRuleStrategy extends PlayRuleStrategy:
71   def calculateWinningPlayer(
72       cardsOnTableByPlayer: List[(PlayerModel, CardModel)]
73   ): Option[PlayerModel] =
74     val firstCardPlayed = cardsOnTableByPlayer.head._2
75     val playedSuit = firstCardPlayed.suit
76     val maxCardByPlayer =
77       cardsOnTableByPlayer
78         .filter(_._2.suit == playedSuit)
79         .max(Ordering.by(_._2.rank))
80     val winningPlayer = cardsOnTableByPlayer.find(_.eq(maxCardByPlayer)).get._1
81     Some(winningPlayer)
